dateFormat 関数を改良して、modeによって、日付だけの表示か、時分秒まで表示するか切り替えられるようにする 
function dateFormat(date, mode) {
  let dateObject = new Date(date);

  const year = dateObject.getFullYear();
  const month = dateObject.getMonth() + 1;
  const day = dateObject.getDate();

  const hour = dateObject.getHours();
  const minute = dateObject.getMinutes();
  const second = dateObject.getSeconds();

  if(mode == 1) {
    return `${year}年${month}月${day}日 ${hour}:${minute}:${second}`;
  } else {
    return month + '月' + day + '日';
  }
}

10未満の数字の頭に 0 をつける関数 add0 を作成して dateFormat 関数の中で 時、分、秒 の頭に0をつける

add0関数

function add0 (val) {
  if(val < 10) {
    val = '0' + val;
  }
  return val;
}

dateFormat関数の中で、時、分、秒 に add0関数をつかって 0 をつける

function dateFormat(date, mode) {
  let dateObject = new Date(date);

  const year = dateObject.getFullYear();
  const month = dateObject.getMonth() + 1;
  const day = dateObject.getDate();

  const hour = add0(dateObject.getHours());
  const minute = add0(dateObject.getMinutes());
  const second = add0(dateObject.getSeconds());

  if(mode == 1) {
    return `${year}年${month}月${day}日 ${hour}:${minute}:${second}`;
  } else {
    return month + '月' + day + '日';
  }
}
