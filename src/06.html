$randomNumber = rand(1, 100); // 1から100のランダムな数を生成
if ($randomNumber <= 23) { // 23%の確率で「大吉」
    return '大吉';
} elseif ($randomNumber <= 33) { // 10%の確率で「中吉」
    return '中吉';
} elseif ($randomNumber <= 46) { // 13%の確率で「小吉」
    return '小吉';
} elseif ($randomNumber <= 70) { // 24%の確率で「吉」
    return '吉';
} elseif ($randomNumber <= 89) { // 19%の確率で「末吉」
    return '末吉';
} elseif ($randomNumber <= 98) { // 9%の確率で「凶」
    return '凶';
} else { // 2%の確率で「大凶」
    return '大凶';
}

  "大吉" => 7,
  "中吉" => 20,
  "吉" => 30,

  $rand = rand(0,5);

  echo $_REQUEST['name'],"さん<br>";
  echo $_REQUEST['age'],"歳のあなたの運勢は<br>";

      echo "大吉です";

関数 (function)

関数とは

関数とは、処理のかたまり、ひとまとまりのプログラムのことです。
関数は再利用でき、名前を付けて一度書いてしまえば次に使うときは名前に基づいて呼び出すだけなので、短い記述で繰り返し同じものを使うことができます。

phpの関数

function 名前(){} のような書式で記述し、{} の中に処理を入れます。
PHPでは通常、行の終わりにセミコロンを付けますが、関数の {} の後ろにセミコロンは不要です。
参考サイト： https://www.php.net/manual/ja/functions.user-defined.php
参考サイト： https://qiita.com/chimayu/items/d4bf6eedbe37d392aa25
関数の定義 
function functionName(){
    // 何らかの
    // 処理を
    // 書く
}
関数の呼び出し 
functionName();  // この1行だけで↑の関数（処理のまとまり）を実行できる

php関数の引数と返り値

関数において入力を受ける変数を引数、関数からの出力を返り値といいます。
引数は丸カッコの中に書く事で、その下の {} の中で使用できるようになり、丸カッコに引数を書かないでその値を使用しようとするとエラーになります。
返り値は俗に、戻り値、リターンなどと呼ばれることもあります。 return $val のように指定し、関数の作りによっては返り値がないものもあります。
function functionName($num){ // ←この丸カッコの中の部分が引数
    // 何らかの処理
    $val = $num + 1; // ←ここで使用している$numが、丸カッコで受けたパラメータ
    return $val; // ←この部分が返り値
}

}

echo tasu(5,7);
work03.php (引数には初期値を指定できる) 
function tasu($a,$b=5) {
  return $a + $b;
}

echo tasu(5);

javascript関数の引数と返り値

関数において入力を受ける変数を引数、関数からの出力を返り値といいます。
引数は丸カッコの中に書く事で、その下の {} の中で使用できるようになり、丸カッコに引数を書かないでその値を使用しようとするとエラーになります。
返り値は俗に、戻り値、リターンなどと呼ばれることもあります。 return val のように指定し、関数の作りによっては返り値がないものもあります。
function functionName(num){ // ←この丸カッコの中の部分が引数
    // 何らかの処理
    val = num + 1; // ←ここで使用している$numが、丸カッコで受けたパラメータ
    return val; // ←この部分が返り値
}

javascript関数の宣言方法のいろいろ

変数に関数を入れられる 
const functionName = function(num){ // ←この丸カッコの中の部分が引数
    // 何らかの処理
    val = num + 1; // ←ここで使用しているnumが、丸カッコで受けたパラメータ
    return val; // ←この部分が返り値
}
functionと書く代わりに => を使った書き方(アロー関数) 
const functionName = (num) => { // ←この丸カッコの中の部分が引数
      // 何らかの処理
      val = num + 1; // ←ここで使用しているnumが、丸カッコで受けたパラメータ
      return val; // ←この部分が返り値
}
work07.html (アロー関数を使った書き方) 
const tasu = (a,b=50) => {
  return a + b;
}

document.write( tasu(5,7) );

おみくじを javascript で、書き直してみる

omikuji.html 
<!DOCTYPE html>
<html lang="ja">
<head>
  <meta charset="UTF-8">
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>おみくじ</title>
</head>
<body>
  <form>
    名前:<input name="name" id="name"><br>
    年齢:<input name="age" id="age"><br>
    <input type="submit" value="占う" id="button">
    <div id="output"></div>
  </form>
  <script>
    const name=document.getElementById("name");
    const age=document.getElementById("age");
    const button=document.getElementById("button");
    const output=document.getElementById("output");

    function uranau() {
      unsei = ['大吉','中吉','小吉','吉','凶','大凶'];
      rand = Math.floor( Math.random() * 6 );
      return unsei[rand];
    }

    function makeMessage(name,age,unsei) {
      let msg = name + "さん<br>";
      msg += age + "歳のあなたの今日の運勢は<br>";
      msg += unsei + "です<br>";

      return msg;
    }

    button.onclick = function(){
      output.innerHTML=makeMessage(name.value,age.value,uranau());
      return false;
    }

  </script>
</body>
</html>
