mkdir.shの改造

mkdir.sh の中身 (黄色の部分を追記)
ここに実行するプログラムのPATHを指定することができる
#!/bin/bash
cd public_html
pwd
mkdir 20240507
cd 20240507
echo hello > index.html
echo hi! >> index.html
プログラムに実行権限を付与
chmod 755 mkdir.sh
プログラムを実行してみる
（今度はファイル名の前に bash と書かなくてもよくなった）
./mkdir.sh
mkdir.sh の中身 (黄色の部分を変更)
20240507を$1に変更することで、実行時に好きなフォルダをつくれるようになる
#!/bin/bash
cd public_html
pwd
mkdir $1
cd $1
echo hello > index.html
echo hi! >> index.html
プログラムを実行してみる
（今度はファイル名の前に bash と書かなくてもよくなった）

Chat-GPTが生成したCプログラム
#include <stdio.h>
#include <stdlib.h>
#include <time.h>
#include <sys/stat.h>
#include <sys/types.h>

int main() {
time_t t = time(NULL);
struct tm *tm = localtime(&t);

char folder_name[20];
sprintf(folder_name, "%d-%02d-%02d", tm->tm_year + 1900, tm->tm_mon + 1, tm->tm_mday);

// フォルダを作成
if (mkdir(folder_name, 0777) == -1) {
perror("mkdir error");
return EXIT_FAILURE;
}

// index.htmlファイルを作成
char file_path[50];
sprintf(file_path, "%s/index.html", folder_name);

FILE *file = fopen(file_path, "w");
if (file == NULL) {
perror("fopen error");
return EXIT_FAILURE;
}

fprintf(file, "<html><head><title>Hi!</title></head><body><h1>hi!</h1></body></html>");

fclose(file);

printf("フォルダ '%s' とファイル '%s' が作成されました。\n", folder_name, file_path);

return EXIT_SUCCESS;
}
